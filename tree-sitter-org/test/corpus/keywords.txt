================
Zeroth and heading special keywords
================

#+TODO: SPECIAL KEYWORD
* Heading
#+TODO: This is treated as a special keyword

---

(document
  (zeroth_section
    (comment)
    (special_keyword
      (keyword_key)
      (keyword_value)))
  (heading
    (stars)
    (plain_text)
    (section
      (special_keyword
        (keyword_key)
        (keyword_value)))))
